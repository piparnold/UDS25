💻Sample Dog-Themed R (tidyverse) Code

library(tidyverse)

dogs <- tibble(
  name = c("Bella", "Charlie", "Luna", "Max"),
  breed = c("Labrador", "Poodle", "Beagle", "Bulldog"),
  treats = c(5, 8, 3, 6)
)

dogs %>%
  filter(treats > 4) %>%
  arrange(desc(treats))
